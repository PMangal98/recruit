SELECT employeeinfo.name as emplName, employeeinfo.employee_id as eeId , employeeInfo.email as emplEmail,job.job_id as JID, job.job_type, application.description FROM employerInfo, job, application, employeeinfo WHERE employerinfo.employer_id = job.employer_id AND application.job_id = job.job_id AND employeeInfo.employee_id = application.employee_id AND employerinfo.employer_id = '{$userId}' and status = 0 AND EXISTS (SELECT jobskill.skill_id FROM jobskill, employeeskill WHERE jobskill.job_id = job.job_id AND employeeskill.employee_id = employerinfo.employer_id AND jobskill.duration < employeeskill.total_duration)
